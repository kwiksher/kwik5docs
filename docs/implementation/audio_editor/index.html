function controller:save(book, page, type, name, model)
  local dst = "App/"..book.."/models/"..page .."/audios/"..type.."/"..name..".json"
  util.mkdir("App", book, "models", page, "audios", type)
  util.saveJson(dst, model)
  return dst
end
